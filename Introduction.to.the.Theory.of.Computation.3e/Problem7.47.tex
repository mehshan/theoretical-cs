\begin{enumerate}[label=\alph*.]
\item $\text{D}_{1}\text{P} = \text{NP}$ and
\item $\text{D}_{i}\text{P} = \{A \ | \ A = B \setminus C \text{ for } B \text{ in NP and } C \text{ in } \text{D}_{i-1}\text{P}\}$. \\ (Here $B \setminus C = B \cap \overline{C}.$)
\end{enumerate}
