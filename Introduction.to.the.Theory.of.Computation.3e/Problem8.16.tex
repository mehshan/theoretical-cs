\begin{problem}{8.16}
Read the definition of $MIN\text{-}FORMULA$ in Problem 7.46.
\end{problem}

\begin{problem}[Part]{a}
Show that $MIN\text{-}FORMULA \in$ PSPACE.
\end{problem}

\begin{proof}
To show that $MIN\text{-}FORMULA \in$ PSPACE, we give a polynomial space algorithm $M$. \\

$M =$ \textquotedblleft On input $\langle \phi \rangle$, where $\phi$ is a Boolean formula:
\begin{enumerate}
\item Repeat for every literal $a$ in $\phi$:
\item \hspace*{0.5cm} Construct a new Boolean formula $\phi_t$ by replacing $a$ with $true$ in $\phi$.
\item \hspace*{0.5cm} Test satisfiablity of $\phi_t$. Recursively construct every possible assignment and evaluate it.
\hspace*{0.5cm} Reuse space for each assignment.
\item \hspace*{0.5cm} Reusing space consumed in step 2, construct a new Boolean formula $\phi_f$ by replacing $a$ \\
\hspace*{0.5cm} with $false$ in $\phi$.
\item \hspace*{0.5cm} Test satisfiablity of $\phi_f$. Reusing space consumed in step 3, repeat step 3 for $\phi_f$.
\item \hspace*{0.5cm} If satisfiability of $\phi_t$ is same as $\phi_f$, then \textit{reject}.
\item \textit{Accept}.\textquotedblright
\end{enumerate}
\end{proof}

\begin{problem}[Part]{b}
Explain why this argument fails to show that $MIN\text{-}FORMULA \in$ coNP: \\
If $\phi \notin MIN\text{-}FORMULA$, then $\phi$ has a smaller equivalent formula. An NTM can verify that $\phi \in \overline{MIN\text{-}FORMULA}$ by guessing that formula.
\end{problem}

\begin{proof}
This argument fails to show that $MIN\text{-}FORMULA \in$ coNP because for $MIN\text{-}FORMULA$ to be in coNP, an NTM must decide $\overline{MIN\text{-}FORMULA}$ in non-deterministic polynomial time.
\end{proof}
